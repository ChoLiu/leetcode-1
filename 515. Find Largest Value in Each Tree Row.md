# 515. Find Largest Value in Each Tree Row

You need to find the largest value in each row of a binary tree.

**Example:**

```
Input: 

          1
         / \
        3   2
       / \   \  
      5   3   9 

Output: [1, 3, 9]
```



## Solution

## Code1

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def largestValues(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        def preorder(root,depth):
            if not root:
                return
            
            if depth == len(res):
                res.append(root.val)
            elif res[depth] < root.val:
                res[depth] = root.val
            preorder(root.left,depth+1)
            preorder(root.right,depth+1)
            
        res = []
        preorder(root,0)
        return res
```



## Code2

```python
# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution(object):
    def largestValues(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        res = []
        stack = []
        stackd = []
        if not root:
            return res
        stack.append(root)
        stackd.append(0)
        while stack:
            node = stack.pop()
            depth = stackd.pop()
            if depth == len(res):
                res.append(node.val)
            else:
                res[depth] = max(res[depth],node.val)
            if node.left:
                stack.append(node.left)
                stackd.append(depth+1)
            if node.right:
                stack.append(node.right)
                stackd.append(depth+1)
        return res
```
